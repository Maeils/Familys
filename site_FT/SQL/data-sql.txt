CREATE NEW DATABASE IF NOT EXISTS ft_database;

CREATE TABLE IF NOT EXISTS user (
	id_user INT NOT NULL AUTO-INCREMENT,
	mail VARCHAR(255) NOT NULL ,
	nom VARCHAR(255) NOT NULL ,
	prenom VARCHAR(255) NOT NULL ,
	pseudo VARCHAR(255) NOT NULL ,
	PRIMARY KEY id_user (id_user)
) ENGINE=InnoDB DEFAULT CHARACTER SET=utf8;

CREATE TABLE IF NOT EXISTS jeu (
	id_jeu INT NOT NULL AUTO-INCREMENT,
	nom VARCHAR(255) NOT NULL ,
	url_image VARCHAR(255) NOT NULL ,
	studio_developpement VARCHAR(255) NOT NULL ,
	description VARCHAR(255) NOT NULL ,
	avis_ft VARCHAR(255) NOT NULL ,
	site_web VARCHAR(255) NOT NULL ,
	PRIMARY KEY id_jeu (id_jeu)
) ENGINE=InnoDB DEFAULT CHARACTER SET=utf8;

CREATE TABLE IF NOT EXISTS personnage (
	id_personnage INT NOT NULL AUTO-INCREMENT,
	pseudo VARCHAR(255) NOT NULL ,
	niveau VARCHAR(255) NOT NULL ,
	id_serveur VARCHAR(255) NOT NULL ,
	PRIMARY KEY id_personnage (id_personnage)
) ENGINE=InnoDB DEFAULT CHARACTER SET=utf8;
